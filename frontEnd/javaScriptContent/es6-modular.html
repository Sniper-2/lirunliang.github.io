    # ES6模块化语法

    # 一.模块的导出

    • ES6导出模块有以下几种方式：

      • module.exports
      • export default
      • export

    • 三种不同的导出方式的区别

      • 1.module.exports不能跟export default、export同时使用，会报错；
      • 2.export default跟export可以同时使用，根据import的方式不同，取到的模块也会不同；
      • 3.通过export导出的模块，import时需要加{}，其他方式则不需要；
      • 4.在一个模块中，export可以有多个，export default跟module.exports仅有一个；
      • 5.export能直接导出变量表达式，export default跟module.exports则不行；

    • 模块的方式以及引入方式

      • 1.module.exports
      // 模块文件 model.js
let testObj = { name: '李润亮' }
module.exports = testObj              // 导出对象
module.exports = { name: '李润亮' }   // 导出对象

// 引入方式 main.js
import testObj from 'model.js';       // testObj --> { name: '李润亮' } 
import { name as mainName } from 'model.js'; // mainName(新的属性命名空间) --> 李润亮
import { name } from 'model.js';      // name --> 李润亮 
const mainObj = require('model.js')   // testObj => { name: '李润亮' } 


// 多个函数导出
// 模块文件 model.js
function a () { console.log('a') };
function b () { console.log('b') };
function c () { console.log('c') };
module.exports = { a, b, c }

// 引入 main.js
import allFun from 'model.js'; // allFun => { a, b, c };
import { a as mainA } from 'model.js';  // mainA(新的方法命名空间) => function a () { console.log('a') };
import { a } from 'model.js';      // a => function a () { console.log('a') };
      • 使用exports导出模块，使用exports导出的模块，import需要使用{}
      // 模块文件 model.js, 可以导出多个
exports function modelFunA () { console.log('modelFunA') };
exports function modelFunB () { console.log('modelFunB') };
exports const test = 10;
exports const modelObj = { objA: 10, ObjB: 20 };

// 引入 main.js
import mainFun from 'model.js';   // undefined,无法导出对应的方法
import modelFunA from 'model.js'; // undefined,无法导出对应的方法

import { modelFunA } from 'model.js'; // modelFunA => { console.log('modelFunA') }
import { modelFunA as mainFunA } from 'model.js'; // mainFunA => { console.log('modelFunA') }
import { test as mainTest } from 'model.js'; // mainTest => 10
      • 使用export default导出模块
      // 模块导出 model.js
export default function () { console.log('aaa') };

// 引入 main.js
import mainFun from 'model.js';   // mainFun => function () { console.log('aaa') };
import { mainFun } from 'model.js';   // mainFun => undefined


// 模块导出 model.js
let modelObj = { a: 10 }
export default modelObj

// 引入 main.js
import mainObj from 'model.js';   // mainObj => { a: 10 }
import { mainObj } from 'model.js';   // mainObj => undefined
      • 使用export default跟exports导出模块
      // model.js
exports function modelFunA () { console.log('modelFunA') };
let modelObj = { a: 10 }
export default modelObj

// main.js
import { modelFunA as mainFun } from 'model.js';   // mainFun => { console.log('modelFunA') }
import mainObj from 'model.js';   // mainObj => { a: 10 }
