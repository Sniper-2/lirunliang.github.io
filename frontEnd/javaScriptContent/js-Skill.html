    # 1. 获取数组中最大值跟最小值
    getMaxAndMin (arrList) {
  return {
    max: Math.max.apply(null, arrList),
    min: Math.min.apply(null, arrList)
  }
}
/**
 * max 跟 min 方法接受一串数值，这串数值里面不能包含非空的字符串，否则会返回NaN 
 * apply不改变max方法this的指向，利用apply第二个参数为一个参数数组的特性，
 * 把我们需要求值的数组当成max的参数传递进去，max就会放回最大值了。
 */

    # 5. 随机生成字符串
    function random_string (len) {
  len = len || 32;
  var chars = 'ABCDEFGHJKMNPQRSTWXYZabcdefhijkmnprstwxyz2345678';
  var maxPos = chars.length;
  var str = '';
  for (let i = 0; i < len; i++) {
    str += chars.charAt(Math.floor(Math.random() * maxPos));
  }
  return str;
}

    # 27.使用Map/Object来解决多重判断
    • 需求： 通过不同类型来调用不同的接口
    // 错误实例,这样会造成代码冗余，十分不简洁，同事复制也不好复制,后续增加多种类型需要再次增加判断
getApiUrl (type) {
  if (type === 1) return 'apiTest1';
  if else (type === 2) return 'apiTest2';
  if else (type === 3) return 'apiTest3';
  if else (type === 4) return 'apiTest4';
  if else (type === 5) return 'apiTest5';
}

// 理想做法
getApiUrl (type) {
  const getAPI = new Map()
      .set(1, 'apiTest1')
      .set(2, 'apiTest2')
      .set(3, 'apiTest3');
      .set(4, 'apiTest4');
      .set(5, 'apiTest5');

  return getAPI.get(type)
}

// 或者
getApiUrl (type) {
  const getAPI = {
    1: 'apiTest1',
    2: 'apiTest2',
    3: 'apiTest3',
    4: 'apiTest4',
    5: 'apiTest5'
  };
  
  return getAPI[type] || [];
}
