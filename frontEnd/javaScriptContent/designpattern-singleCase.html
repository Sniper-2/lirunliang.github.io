    # 前端设计模式 - 单例模式

    • 定义：单例模式是保证一个类只有一个实例，并且提供一个访问它的全局访问点；

    • 需求：前端对api接口数据进行本地内存缓存；

    • 特性：可以被实例化，且只能实例化一次，再次实例化生成的也是第一个实例；

    • 应用：当你将问题抽象之后，发现解决方案具有单例模式的特性，那么就能使用单例模式去解决问题

    • 例子 - 利用单例模式对数据进行本地缓存

    
class ApiDataCache {
  constructor () {
    this.cacheData = {};            // 缓存数据地对象
    this.maxCacheNum = 100;         // 最大缓存条数
  }

  // 静态方法，用来初始化改单例模式对象，使其具有‘可以被实例化，且只能实例化一次，再次实例化生成的也是第一个实例’特性
  static initCacheData () {
    if (!ApiDataCache.instance) {
      ApiDataCache.instance = new ApiDataCache();
      return ApiDataCache.instance
    }
    return ApiDataCache.instance;
  }


  /**
   * 对数据进行缓存
   * @param {String} key      缓存到cache对象中的key值
   * @param {*} data          进行缓存的数据
   * @param {Number} times    该组缓存的有效期
   */
  addCacheData (key, data, times = 300) {

    // 写入缓存之前先判断目前缓存的接口数量是否超过预设的阈值，防止本地缓存过多
    if (Object.keys(this.cacheData).length > this.maxCacheNum) {
      this.cacheData = {}
		}

    let cacheData = JSON.stringify(data) // 缓存之前需要对数据序列化，因为这个涉及到引用数据类型的特性
    cacheData = JSON.parse(cacheData)

    let dataItem = {
      ...cacheData,
      times,
      dateTimes: new Date().getTime()
    }
    this.cacheData[key] = dataItem
  }

  // 获取数据缓存
  getCacheData (key) {

    // 先判断要进行获取的数据是否为有效数据
    if (!this.isInvalidData(key)) return false

    // 深拷贝数据，并返回
    let cacheData = JSON.stringify(this.cacheData[key])
    return JSON.parse(cacheData)
  }

  /**
   * 对数据进行更新
   * @param {String} key      需要更新的cache对象中的key值
   * @param {*} data          新的数据
   * @param {Number} times    该组缓存的有效期，默认是5分钟，单位是秒
   */
  upCacheData (key, data, times = 300) {
    this.addCacheData(key, data, times)
  }

  // 删除缓存
  delCache (key) {
    delete this.cacheData[key]
    return true;
  }

  // 查看缓存
  lookCacheData () {
    return this.cacheData
  }


  // 判断即将读取的数据是否为失效数据
  isInvalidData (key) {
    // 缓存中是否存在该数据
    if (!this.cacheData[key]) return false

    // 数据是否超过了缓存有效期
    let cacheData = this.cacheData[key];
    let times = cacheData.times;
    let cacheStartTimes = cacheData.dateTimes;
    let nowDateTimes = new Date().getTime()
    let afterTimes = (nowDateTimes - cacheStartTimes ) / 1000

    // 当前时间大于该数据缓存的时候的时间，表示数据已经过期了
    if (afterTimes > times) {
      this.delCache(key)
      return false
    }

    return true
  }

}

module.exports = ApiDataCache.initCacheData()
